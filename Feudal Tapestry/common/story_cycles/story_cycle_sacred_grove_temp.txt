# Story cycle used by the Sponsor Gladiatorial Games decision
# Get four events (one per week), and then end it

story_visit_sacred_grove = {
	on_setup = {
		story_owner = {

			var:county_grove = {
				save_scope_as = county
			}
			# Need to do this in case the modifier disappears before the event triggers
			if = {
				limit = {
					scope:county = {
						has_county_modifier = sacred_grove_level_3
					}
				}
				add_character_flag = sacred_grove_level_3
			}
			else_if = {
				limit = {
					scope:county = {
						has_county_modifier = sacred_grove_level_2
					}
				}
				add_character_flag = sacred_grove_level_2
			}
			# Failsafe
			else = {
				add_character_flag = sacred_grove_level_1
			}
			trigger_event = {
				id = bryten_decision.0017 
				days = { 7 21 }
			}
		}
	}

	on_end = {
		story_owner = {
			if = {
				limit = {
					has_character_flag = planning_an_activity
				}
				remove_character_flag = planning_an_activity
			}
			if = {
				limit = {
					has_character_flag = sacred_grove_level_3
				}
				remove_character_flag = sacred_grove_level_3
			}
			if = {
				limit = {
					has_character_flag = sacred_grove_level_2
				}
				remove_character_flag = sacred_grove_level_2
			}
			if = {
				limit = {
					has_character_flag = sacred_grove_level_1
				}
				remove_character_flag = sacred_grove_level_1
			}

			if = {
				limit = {
					has_character_flag = sacred_grove_level_3
				}
				remove_character_flag = sacred_grove_level_3
			}
			if = {
				limit = {
					has_character_flag = sacred_grove_level_2
				}
				remove_character_flag = sacred_grove_level_2
			}
			if = {
				limit = {
					has_character_flag = sacred_grove_level_1
				}
				remove_character_flag = sacred_grove_level_1
			}
			remove_variable = county_grove
		}	
	}

	# Cancel the games
	on_owner_death = {
		end_story = yes
	}
}
