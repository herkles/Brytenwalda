
bryten_bible_translation_story = { #Used for the Scholarship progress event
	on_setup = {
		set_variable = {
			name = bryten_bible_translation_quality
			value = 0
		}
		set_variable = {
			name = duration_left
			value = 5
		}
	}
	on_end = {
		story_owner = {
			remove_character_modifier = bryten_bible_translation_ongoing_modifier
		}
	}
	on_owner_death = {
		end_story = yes
	}
	effect_group = {
		days = { 30 90 }

		first_valid = {
			triggered_effect = {
				trigger = {
					var:duration_left < 1
				}
				effect = {
					#Story resolution
					if = {
						limit = {
							var:bryten_bible_translation_quality >= 5
						}
						story_owner = {
							trigger_event = bryten_heresy.510
						}
					}
					else_if = {
						limit = {
							var:bryten_bible_translation_quality < 2
						}
						story_owner = {
							trigger_event = bryten_heresy.511
						}
					}
					else = {
						random_list = {
							1 = {
								compare_modifier = {
									value = var:bryten_bible_translation_quality
								}
								story_owner = {
									trigger_event = bryten_heresy.510
								}
							}
							3 = {
								compare_modifier = {
									value = var:bryten_bible_translation_quality
									multiplier = -1
									min = -2
								}
								story_owner = {
									trigger_event = bryten_heresy.510
								}
							}
						}
					}
				}
			}
			triggered_effect = {
				trigger = { always = yes }
				effect = {
					if = {
						limit = {
							NOT = { exists = var:had_first_event }
						}
						change_variable = {
							name = duration_left
							add = -1
						}
						set_variable = {
							name = had_first_event
							value = yes
						}
						story_owner = {
							trigger_event = {
								on_action = bryten_bible_translation_story_events
							}
						}
					}
					else = {
						random_list = {
							1 = {
								change_variable = {
									name = duration_left
									add = -1
								}
							}
							3 = {
								trigger = { var:bryten_bible_translation_quality < 5 }
								change_variable = {
									name = duration_left
									add = -1
								}
								story_owner = {
									trigger_event = {
										on_action = bryten_bible_translation_story_events
									}
								}
							}
						}
					}
				}
			}
		}
	}
}
