bryten_england.030 = {# Anglo-Saxon Culture forming
	hidden = yes
	
	immediate = {
		# Save scopes for player notification events.
		culture:northumbrian = { save_scope_as = northumbrian }
		culture:mercian = { save_scope_as = mercian }
		culture:west_saxon = { save_scope_as = west_saxon }
		culture:east_anglian = { save_scope_as = east_anglian }


		culture:anglo_saxon = {
			# 'Hybridize' this culture by copying innovations from parent cultures. Leave traditions as defined in the database.
			reset_culture_creation_date = yes
			get_all_innovations_from = culture:mercian
			get_all_innovations_from = culture:northumbrian
			get_all_innovations_from = culture:west_saxon
			get_all_innovations_from = culture:east_anglian

			# Save scope for player notification events.
			save_scope_as = anglo_saxon
		}

		# All Norse counties in England will convert to Anglo-saxon
		culture:northumbrian = {
			every_culture_county = {
				limit = {
					title_province = {
						geographical_region = custom_england
					}
				}
				add_to_list = anglo_saxon_counties
			}
		}
		culture:mercian = {
			every_culture_county = {
				limit = {
					title_province = {
						geographical_region = custom_england
					}
				}
				add_to_list = anglo_saxon_counties
			}
		}
		culture:west_saxon = {
			every_culture_county = {
				limit = {
					title_province = {
						geographical_region = custom_england
					}
				}
				add_to_list = anglo_saxon_counties
			}
		}
		culture:east_anglian = {
			every_culture_county = {
				limit = {
					title_province = {
						geographical_region = custom_england
					}
				}
				add_to_list = anglo_saxon_counties
			}
		}

		# Convert the england counties to anglo-norse
		every_in_list = {
			list = anglo_saxon_counties
			limit = {
				# Same-culture players get an event informing them of their county's conversion *before* it happens (for their provinces only).
				NOT = {
					holder = {
						OR = {
							AND = {
								is_ai = no
								culture = { has_cultural_pillar = heritage_west_germanic }
								capital_province = { geographical_region = custom_england }
							}
							any_liege_or_above = {
								is_ai = no
								culture = { has_cultural_pillar = heritage_west_germanic }
								capital_province = { geographical_region = custom_england }
							}
						}
					}
				}
			}
						
			# All England counties in England become anglo_saxon_counties
			convert_county_and_holder_if_in_region_effect = {
				OLD_CULTURE = culture:northumbrian
				NEW_CULTURE = culture:anglo_saxon
				REGION = custom_england
			}
			convert_county_and_holder_if_in_region_effect = {
				OLD_CULTURE = culture:mercian
				NEW_CULTURE = culture:anglo_saxon
				REGION = custom_england
			}
			convert_county_and_holder_if_in_region_effect = {
				OLD_CULTURE = culture:west_saxon
				NEW_CULTURE = culture:anglo_saxon
				REGION = custom_england
			}
			convert_county_and_holder_if_in_region_effect = {
				OLD_CULTURE = culture:east_anglian
				NEW_CULTURE = culture:anglo_saxon
				REGION = custom_england
			}
		}

		# Rulers get the next decision
		every_ruler = {
			limit = {
				culture = { has_cultural_pillar = heritage_west_germanic }
				is_landed = yes
				is_independent_ruler = yes
				not = {culture = culture:anglo_norse }
				capital_province = { geographical_region = custom_england }
			}
			trigger_event = bryten_england.031
		}

		# Send the appropraite notification event to each player.
		every_player = {
			if = {
				limit = {
					is_ai = no
					save_temporary_scope_as = player
					any_in_list = {
						list = anglo_saxon_counties
						holder = {
							is_within_diplo_range = { CHARACTER = scope:player }
						}
					}
				}
				trigger_event = bryten_england.032
			}
		}
	}
}

		culture:northumbrian = { save_scope_as = northumbrian }

			# 'Hybridize' this culture by copying innovations from parent cultures. Leave traditions as defined in the database.
			reset_culture_creation_date = yes
			get_all_innovations_from = culture:danish
			get_all_innovations_from = culture:northumbrian

			# Save scope for player notification events.
			save_scope_as = anglo_norse
		}

		# All Norse counties in England will convert to Anglo-Norse
		culture:danish = {
			every_culture_county = {
				limit = {
					title_province = {
						geographical_region = custom_england
					}
				}
				add_to_list = anglo_norse_counties
			}
		}

		culture:norwegian = {
			every_culture_county = {
				limit = {
					title_province = {
						geographical_region = custom_england
					}
				}
				add_to_list = anglo_norse_counties
			}
		}

		# Convert the norse counties to anglo-norse
		every_in_list = {
			list = anglo_norse_counties
			limit = {
				# Same-culture players get an event informing them of their county's conversion *before* it happens (for their provinces only).
				NOT = {
					holder = {
						OR = {
							AND = {
								is_ai = no
								culture = { has_cultural_pillar = heritage_north_germanic }
								capital_province = { geographical_region = custom_england }
							}
							any_liege_or_above = {
								is_ai = no
								culture = { has_cultural_pillar = heritage_west_germanic }
								capital_province = { geographical_region = custom_england }
							}
						}
					}
				}
			}
						
			# All Norse counties in England become anglo_norse
			convert_county_and_holder_if_in_region_effect = {
				OLD_CULTURE = culture:norse
				NEW_CULTURE = culture:anglo_norse
				REGION = custom_england
			}
			convert_county_and_holder_if_in_region_effect = {
				OLD_CULTURE = culture:danish
				NEW_CULTURE = culture:anglo_norse
				REGION = custom_england
			}
			convert_county_and_holder_if_in_region_effect = {
				OLD_CULTURE = culture:norwegian
				NEW_CULTURE = culture:anglo_norse
				REGION = custom_england
			}
		}

		# Rulers get the next decision
		every_ruler = {
			limit = {
				AND = {
					culture = { has_cultural_pillar = heritage_north_germanic }
					is_landed = yes
					OR = {# Must own one of the Kingdoms
						has_title = title:k_northumbria
						has_title = title:k_mercia
						has_title = title:k_wessex
						has_title = title:k_east_anglia
					}				
					capital_province = { geographical_region = custom_england }
				}
			}
			trigger_event = bryten_england.041
		}

		# Send the appropraite notification event to each player.
		every_player = {
			if = {
				limit = {
					is_ai = no
					save_temporary_scope_as = player
					any_in_list = {
						list = anglo_norse_counties
						holder = {
							is_within_diplo_range = { CHARACTER = scope:player }
						}
					}
				}
				trigger_event = bryten_england.042
			}
		}
	}
}
bryten_england.041 = {# Norse Ruled Realms
	type = character_event
	title = bryten_england.041.t
	desc = bryten_england.041.desc
	theme = culture_change

	trigger = {# safety checks
	culture = { has_cultural_pillar = heritage_north_germanic }
		is_landed = yes
		OR = {# Must own one of the Kingdoms
			has_title = title:k_northumbria
			has_title = title:k_mercia
			has_title = title:k_wessex
			has_title = title:k_east_anglia
		}				
		capital_province = { geographical_region = custom_england }
	}

	immediate = {
		if = {# For Northumbria
			limit = { 
				has_title = title:k_northumbria

			}

				limit = { 	
					is_coastal_county = yes

				add_to_list = northumbrian_counties
			}

				# All northumbrian counties in England become anglo_norse
				convert_county_and_holder_if_in_region_effect = {
					OLD_CULTURE = culture:northumbrian
					NEW_CULTURE = culture:anglo_norse
					REGION = custom_england
				}
			}
		}
		if = {# For Mercia
			limit = { 
				has_title = title:k_mercia 
				culture = { has_cultural_pillar = heritage_north_germanic }
			}
			# All coastal Mercian counties in England will convert to Anglo-Norse
			every_sub_realm_county = {
				limit = { 	
					is_coastal_county = yes
					title_province = { geographical_region = custom_england }
					culture = culture:mercian				
				}
				add_to_list = mercian_counties
			}
			# Convert the Mercian counties to anglo-norse
			every_in_list = {
				list = mercian_counties						
				# All mercian counties in England become anglo_norse
				convert_county_and_holder_if_in_region_effect = {
					OLD_CULTURE = culture:mercian
					NEW_CULTURE = culture:anglo_norse
					REGION = custom_england
				}
			}
		}
		if = {# For East Anglia
			limit = { 
				has_title = title:k_east_anglia 
				culture = { has_cultural_pillar = heritage_north_germanic }
			}

			# All coastal East Anglian counties in England will convert to Anglo-Norse
			every_sub_realm_county = {
				limit = { 	
					is_coastal_county = yes
					title_province = { geographical_region = custom_england }
					culture = culture:east_anglian				
				}
				add_to_list = east_anglian_counties
			}
			# Convert the East Anglian counties to anglo-norse
			every_in_list = {
				list = east_anglian_counties						
				# All east_anglian counties in England become anglo_norse
				convert_county_and_holder_if_in_region_effect = {
					OLD_CULTURE = culture:east_anglian
					NEW_CULTURE = culture:anglo_norse
					REGION = custom_england
				}
			}
		}
		if = {# For Wessex
			limit = {
				has_title = title:k_wessex
				culture = { has_cultural_pillar = heritage_north_germanic }
			}
			# All coastal west_saxon counties in England will convert to Anglo-Norse

				limit = { 	
					is_coastal_county = yes

			CONVERTER = root
			OLD_CULTURE = root.culture
			NEW_CULTURE = culture:anglo_norse
		}
	}
	option = {# Embrace Anglo Norse and catholicism
		name = bryten_england.041.embrace

			CONVERTER = root
			OLD_CULTURE = root.culture
			OLD_FAITH = root.faith
			NEW_CULTURE = culture:anglo_norse
			NEW_FAITH = faith:catholic
		}
	}

	create_dynamic_title = {
		tier = kingdom
		name = {
			first_valid = {
