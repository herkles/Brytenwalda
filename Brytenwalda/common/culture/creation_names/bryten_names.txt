mercian = {# Mercian
	trigger = {
		not = {	exists = global_var:bryten_anglo_saxon_created } # don't get recreated after Anglo-Saxon is formed
		OR = {
			capital_county = {
				title:d_hereford = {
					is_de_jure_liege_or_above_target = prev
				}
			}
			has_primary_title = title:d_hereford
		}
		culture = culture:anglo_saxon
		any_sub_realm_county = {
			culture = culture:anglo_saxon
			duchy = title:d_hereford
		}
	}
}
