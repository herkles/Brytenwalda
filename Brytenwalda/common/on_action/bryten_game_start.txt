#######################################
# Game Start Stuff
#######################################
on_game_start_after_lobby = {
	on_actions = {
		bryten_setup_after_game_start_on_actions
	}
}

bryten_setup_after_game_start_on_actions = {
	effect = {
		character:140 = {# Norman is King in the Norman Conquest Bookmark
			if = { limit = { game_start_date <= 1066.12.25 } }
			trigger_event = bryten_england.001
		}
		character:33358 = {# Aethelred designates Alfred
			if = { limit = { game_start_date <= 867.1.1 } }
			trigger_event = bryten_england.500
		}		


		if = {# setting up the pilgrimage stuff for 1066 start date
			limit = { 
				or = {
					game_start_date <= 1066.12.25
					game_start_date <= 1066.9.15
				}
			}
			set_global_variable = {
				name = bryten_bury_saint_edmunds
				value = yes
			}
			set_global_variable = {
				name = bryten_monks_cuthbert_durham
				value = yes
			}
		}

		if = {# Anglo-Saxons like the Scandivians more due to ties.
			limit = { game_start_date <= 1066.9.15 }
			culture:anglo_saxon = {
				change_cultural_acceptance = {
					target = culture:swedish
					value = 50
					desc = cultural_acceptance_historical_relations
				}
				change_cultural_acceptance = {
					target = culture:norwegian
					value = 50
					desc = cultural_acceptance_historical_relations
				}
				change_cultural_acceptance = {
					target = culture:danish
					value = 50
					desc = cultural_acceptance_historical_relations
				}
				change_cultural_acceptance = {
					target = culture:anglo_norse
					value = 50
					desc = cultural_acceptance_historical_relations
				}
			}
		}
		if = {# Normans Don't like the Anglo Cultures
			limit = { game_start_date = 1066.12.25 }
			culture:norman = {
				change_cultural_acceptance = {
					target = culture:anglo_saxon
					value = -75
					desc = cultural_acceptance_historical_relations
				}
				change_cultural_acceptance = {
					target = culture:anglo_norse
					value = -75
					desc = cultural_acceptance_historical_relations
				}

			}
		}		
	}
}
