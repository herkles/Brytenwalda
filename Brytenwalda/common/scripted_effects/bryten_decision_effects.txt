
unite_kingdom_england_decision_effects = {
	# Create the title and make it primary
	create_title_and_vassal_change = {
		type = created
		save_scope_as = title_change
		add_claim_on_loss = no
	}
	title:k_england = {
		change_title_holder = {
			holder = root
			change = scope:title_change
		}
	}
	resolve_title_and_vassal_change = scope:title_change
	set_primary_title_to = title:k_england

	title:k_kent = { add_to_list = english_kingdoms }
	title:k_northumbria = { add_to_list = english_kingdoms }
	title:k_mercia = { add_to_list = english_kingdoms }
	title:k_east_anglia = { add_to_list = english_kingdoms }
	title:k_wessex = { add_to_list = english_kingdoms }

	every_in_list = {
		list = english_kingdoms
		every_in_de_jure_hierarchy = {
			limit = {
				tier = tier_kingdom
			}
			set_de_jure_liege_title = title:k_england
		}
	}

	hidden_effect = {
		every_in_list = {
			list = english_kingdoms
			limit = {
				NOT = { this = scope:england_kingdom_uniter.primary_title }
				scope:england_kingdom_uniter = { completely_controls = prev }
			}

			#De jure shift all held spanish titles into your primary title
			every_in_de_jure_hierarchy = {
				limit = { tier = tier_duchy }
				set_de_jure_liege_title = scope:england_kingdom_uniter.primary_title
			}

			#Destroys them all!
			scope:england_kingdom_uniter = { destroy_title = prev }
		}
	}
}
